美团开源动态线程池

Gordon 2023-5-8 线程池设计spring

# 由来

没有一个合适的公式，可以帮助我们合理的设定线程池参数，故而需要在实际运行过程中动态调整参数，缩短程序恢复的时间。

官网：https://dynamictp.cn (opens new window)

gitee地址：https://gitee.com/dromara/dynamic-tp (opens new window)

github地址：https://github.com/dromara/dynamic-tp (opens new window)

# 轻松入门

  1. 引入相应配置中心的依赖，maven 依赖见下述
  2. 配置中心配置线程池实例，配置文件见下述
  3. 启动类加 @EnableDynamicTp 注解
  4. 使用 @Resource 或 @Autowired 进行依赖注入，或通过 DtpRegistry.getDtpExecutor("name") 获取
  5. 通过以上 4 步就可以使用了，是不是感觉超级简单呀

package com.gordon.common;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.context.annotation.Bean;
import org.springframework.core.env.Environment;

import java.util.concurrent.ExecutorService;
import java.util.concurrent.LinkedBlockingQueue;
import java.util.concurrent.ThreadPoolExecutor;
import java.util.concurrent.TimeUnit;

@SpringBootApplication
public class MyApplication {

    @Autowired //引用配置的环境参数
    public Environment environment;

    // CommonExecutorsController 需要提前创建注入
    @Bean //作为一个bean初始化
    public ExecutorService executorService(){
        Integer corePoolSize = Integer.valueOf(environment.getProperty("dtp.core-pool-size"));
        Integer maximumPoolSize = Integer.valueOf(environment.getProperty("dtp.maximum-pool-size"));

        //打印确认配置是否正确注入
        System.out.println("corePoolSize = " + corePoolSize);

        return new ThreadPoolExecutor(corePoolSize, maximumPoolSize,
                0L, TimeUnit.MILLISECONDS,
                new LinkedBlockingQueue<Runnable>());
    }

    public static void main(String[] args) {
        SpringApplication.run(MyApplication.class,args);
    }
}

创建一个配置类，并监听

package com.gordon.singleDtp;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.core.env.Environment;

@Configuration
public class AutoSingleDtpConfiguration  {//AutoSingleDtpConfiguration 这个要生效还要配置spring.factories

    @Autowired //引用配置的环境参数
    public Environment environment;

    // CommonExecutorsController 需要提前创建注入
    @Bean //作为一个bean初始化
    public DtpExecutor executor(){
        Integer corePoolSize = Integer.valueOf(environment.getProperty("dtp.core-pool-size"));
        Integer maximumPoolSize = Integer.valueOf(environment.getProperty("dtp.maximum-pool-size"));

        return new DtpExecutor(corePoolSize,maximumPoolSize);
    }

    @Bean
    public NacosListener nacosListener(){
        return new NacosListener();
    }
}

创建监听器

package com.gordon.singleDtp;

import com.alibaba.nacos.api.annotation.NacosInjected;
import com.alibaba.nacos.api.config.ConfigService;
import com.alibaba.nacos.api.config.listener.Listener;
import org.springframework.beans.factory.InitializingBean;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.config.YamlPropertiesFactoryBean;
import org.springframework.core.io.ByteArrayResource;

import java.util.Properties;
import java.util.concurrent.Executor;
import java.util.concurrent.Executors;

public class NacosListener implements Listener, InitializingBean {

    @NacosInjected
    private ConfigService configService;

    @Autowired
    private DtpExecutor executor;

    @Override //启动一个后台进程去执行内容的更新
    public Executor getExecutor() {
        return Executors.newFixedThreadPool(1);
    }

    @Override
    public void receiveConfigInfo(String content) {

        YamlPropertiesFactoryBean bean = new YamlPropertiesFactoryBean();
        bean.setResources(new ByteArrayResource(content.getBytes()));
        Properties properties = bean.getObject();
        Integer corePoolSize = Integer.valueOf(properties.getProperty("dtp.core-pool-size"));
        Integer maximumPoolSize = Integer.valueOf(properties.getProperty("dtp.maximum-pool-size"));
        executor.setCorePoolSize(corePoolSize);
        executor.setMaximumPoolSize(maximumPoolSize);

    }

    @Override
    public void afterPropertiesSet() throws Exception {
        configService.addListener("dtp.yaml","DEFAULT_GROUP",this);
    }
}
